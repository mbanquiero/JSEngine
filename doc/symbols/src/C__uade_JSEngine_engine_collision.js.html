  3 /**
  4  * @fileoverview collision.js
  5 <P> Implementa colisiones entre 2 cuerpos</P>
  6 <P>  Fase de Collision Detection</P> 
  7 <P> Los motores de fisicas usualmente tienen 2 etapas bien definidas</P>
  8 <P> * Etapa de Collision Detection: el objetivo de esta etapa es determinar si 2 o mas cuerpos entran en colision. </P>
  9 <P> PAra ello se aplican una serie de algoritmos geometricos bastante estudiados. </P>
 10 <P> Esta etapa a su vez se puede dividir en 2 fases :</P>
 11 <P>              - Broad Phase. Se trata de rapidamente descartar aquellos pares de cuerpos que por estar lo suficienemente
 12                               alejados no pueden entrar en colision. Se suelen usar bounding box y otro tipo de simplificaciones
 13                               y distintas estructuras jerarquicas. La idea es que rapidamente se puedan detectar pares de 
 14                               candidatos que pueden entrar en colision.</P>
 15 							  
 16 <P>              - Narrow Phase. Con el resultado de pares "posibles" de la fase anterior, se determinar exactamente si 
 17                               esos "candidatos" a colision, realmente estaban colisionando, y en caso afirmativo
 18                               colectar informacion especifica de la colision, que luego va a servir en etapas posteriores
 19                               como el punto exacto de la colision, la normal, y la penetracion.</P>
 20 							  
 21 <P> * Etapa de Collision Response : una vez dectectada la collision, el motor tiene que tomar alguna acción para evitar
 22  que los cuerpos que collisionaron siguen en esa situacion. En nuestro caso se resuelve aplicando un impulso de tal forma
 23  que los cuerpos se alejen en la direccion que chocaron,  y asi evitar que sigan en colision = Impulse Resolution.</P>
 24  
 25 
 26 <P> En este motor no tenemos Broad Phase, los algoritmos que se codifican a continuacion, corresponden a la llamada
 27  Narrow Phase, y tienen por objetivo determianr si hay o no colision, y en casi que asi fuera computar:</P>
 28  
 29 <P> i) punto de contacto</P>
 30 <P> ii) direccion normal de la colision</P>
 31 <P> iii) penetracion</P>
 32 <P> puede que los cuerpos entre en colision en mas de un punto, por eso la informacion de contacto tiene un array 
 33 y una cantidad de contactos.</P>
 34 
 35 */

 41 /**
 42  * Colision entre esferas
 43  * @function CircleCircle
 44  * @param {Body} a esfera A 
 45  * @param {Body} b esfera B 
 46  * @param {ContactSolver} contact_info informacion de contacto
 47  */
 48 function CircleCircle(a, b ,contact_info)
 49 {
 50     var A = a.shape;            
 51     var B = b.shape;
 52 
 53     var normal = vec2_substract(b.position , a.position);
 54     var dist_sqr = normal.lengthSqr();
 55     var radius = A.radius + B.radius;
 56 
 57     // No estan en contacto
 58     if(dist_sqr >= radius * radius)
 59     {
 60         contact_info.contact_count = 0;
 61         return false;
 62     }
 63 
 64     var distance = Math.sqrt(dist_sqr);
 65     contact_info.contact_count = 1;
 66 
 67     if(Math.abs(distance) < EPSILON)
 68     {
 69         // Estan tocando justo
 70         contact_info.penetration = A.radius;
 71         contact_info.normal = new Vector2( 1, 0 );
 72         contact_info.contacts[0] = a.position;
 73     }
 74     else
 75     {
 76         contact_info.penetration = radius - distance;
 77         contact_info.normal = normal.multiply(1/distance);
 78         contact_info.contacts[0] = vec2_add(vec2_multiply(normal, A.radius), a.position);
 79     }
 80     return true;
 81 }

 90 /**
 91  * Colision entre esfera y Poligono
 92  * Para esfera contra poligono se usa una adaptacion del SAT
 93  * @function CirclePolygon
 94  * @param {Body} a esfera a 
 95  * @param {Body} b poligono b 
 96  * @param {ContactSolver} contact_info informacion de contacto
 97  */
 98 function CirclePolygon(a, b, contact_info)
 99 {
100     var A = a.shape;            // circulo
101     var B = b.shape;            // poligono
102 
103     contact_info.contact_count = 0;
104 
105     // Transoformo todo a al espacio del poligono
106     // Para transformar un vector del world space al polygon B space, tengo que restarle la posicion del CM del poligono B
107     // Y luego multiplicarlo por la matriz de transformacion inversa del B (que es igual a la transpuesta, por ser una matriz
108     // de rotacion).
109 
110     // Demostracion
111     // Los puntos en model space (o poligon space) se transforman a world space asi:
112     // (subindice ws = world space)
113     // Pws = B.matWorld*P + B.pos
114     // luego si estoy al reves, y tengo Pws y quiero P tengo que despejar
115     // P = inversa(B.matWorld) *(Pws - B.pos)
116     // como es una matriz de rotacion, su inversa es igual a su transpuesta
117     // P = transpuesta(B.matWorld) *(Pws - B.pos)
118 
119     var center = vec2_transform(B.matWorld.Transpose()  , vec2_substract(a.position , b.position));
120 	
121     // Calcular el lado donde se produce la minima penetracion
122     var separation = -1000000;
123     var faceNormal = 0;
124     for(var i = 0; i < B.m_vertexCount; ++i)
125     {
126 		var s = dot( B.normals[i], vec2_substract(center , B.vertices[i]));
127 		if(s > A.radius)
128 			return false;
129 
130 		if(s > separation)
131         {
132 			separation = s;
133 			faceNormal = i;
134         }
135     }
136 
137     // Adaptacion de algoritmo SAT 
138     var v1 = B.vertices[faceNormal];
139     var v2 = B.vertices[faceNormal+1];
140 
141     // Verifico si el el punto center esta dentro del poligono
142     if(separation < EPSILON)
143     {
144         // El punto esta dentro del poligono
145         contact_info.contact_count = 1;
146         contact_info.normal = vec2_multiply(vec2_transform(B.matWorld , B.normals[faceNormal]) , -1);
147         contact_info.contacts[0] = vec2_add(vec2_multiply(contact_info.normal , A.radius ) , a.position);
148         contact_info.penetration = A.radius;
149         return true;
150     }
151 
152     var dot1 = dot( vec2_substract(center , v1), vec2_substract(v2, v1));
153     var dot2 = dot( vec2_substract(center , v2), vec2_substract(v1 ,v2));
154     contact_info.penetration = A.radius - separation;
155 
156     // Esta cerca del punto V1
157     if(dot1 <= 0.)
158     {
159         
160         if(vec2_substract(center,v1).lengthSqr() > A.radius * A.radius)
161             return false;
162 
163         contact_info.contact_count = 1;
164         var n = vec2_substract(v1 , center);
165         n = vec2_transform(B.matWorld , n);
166         n.normalize();
167         contact_info.normal = n;
168         v1 = vec2_add(vec2_transform(B.matWorld ,v1) , b.position);
169         contact_info.contacts[0] = v1;
170     }
171 
172     // Esta cerca del punto V2
173     else if (dot2 <= 0.)
174     {
175         if(vec2_substract(center,v2).lengthSqr() > A.radius * A.radius)
176             return false;
177 
178         contact_info.contact_count = 1;
179         var n = vec2_substract(v2, center);
180         v2 = vec2_add(vec2_transform(B.matWorld ,v2) , b.position);
181         contact_info.contacts[0] = v2;
182         n = vec2_transform(B.matWorld ,n);
183         n.normalize( );
184         contact_info.normal = n;
185     }
186     else
187     // en un punto medio entre V1 y V2
188     {
189         var n = B.normals[faceNormal];
190         if(dot( vec2_substract(center ,v1), n ) > A.radius)
191             return false;
192 
193         n = vec2_transform(B.matWorld , n);
194         contact_info.normal = vec2_multiply(n,-1);
195         contact_info.contacts[0] = vec2_add(vec2_multiply(contact_info.normal ,  A.radius) , a.position);
196         contact_info.contact_count = 1;
197     }
198 
199     return true;
200 }

202 // 2D Polygon Collision Detection
203 // http://www.codeproject.com/Articles/15573/D-Polygon-Collision-Detection
204 // http://en.wikipedia.org/wiki/Hyperplane_separation_theorem
205 // http://back2basic.phatcode.net/?Issue_%231:2D_Convex_Polygon_Collision_using_SAT
206 function getSeparationAxis( faceIndex, a, b)
207 {
208     A = a.shape;
209     B = b.shape;
210     var max = -100000;
211 
212     for(var i = 0; i < A.m_vertexCount; ++i)
213     {
214         // Retrieve a face normal from A
215 		var n = A.normals[i];
216 		var nw = vec2_transform(A.matWorld ,  n);
217 
218         // Transformo al espacio local de B
219 		var bT = B.matWorld.Transpose();
220 		n = vec2_transform(bT ,  nw);
221 
222         // maxima distancia de B sobre la direccion -n
223 		var s = B.getMaxProjection(vec2_multiply(n, -1));
224 
225         // Transformo al espacio local de B
226 		var v = A.vertices[i];
227 		v = vec2_add(vec2_transform(A.matWorld , v) , a.position);
228 		v.substract(b.position);
229 		v = vec2_transform(bT , v);
230 
231         // distancia de penetration distance
232 		var d = dot( n, vec2_substract(s , v) );
233 
234 		if (d > max)
235         {
236 		    max = d;
237 		    faceIndex.x = i;
238         }
239     }
240     return max;
241 }

243 // nomenclatura:
244 // Inicidente = es el poligono que esta chocando = IncBody
245 // Reference = es el poligono chocado = RefBody
246 // Es una cuestion de puntos de vista, se dice inicidente y referente, porque se transforma todas las coordenadas
247 // al espsacio del referente, y en ese espacio, el poligono esta "quieto", Luego la velocidad del incidente se computa
248 // de acuerdo a la velocidad relativa en ese marco de referencia.

250 function getIncFace( v, RefBody, IncBody, refIndex )
251 {
252 	var RefPoly = RefBody.shape;
253     var IncPoly = IncBody.shape;
254 	
255     var refNormal = RefPoly.normals[refIndex];
256 
257     // Calculate normal in incident's frame of reference
258     refNormal = vec2_transform(RefPoly.matWorld , refNormal);               // To world space
259     refNormal = vec2_transform(IncPoly.matWorld.Transpose( ) , refNormal);  // To incident's model space
260 
261     // Find most anti-normal face on incident polygon
262     var incFace = 0;
263     var minDot = 100000;
264     for(var i = 0; i < IncPoly.m_vertexCount; ++i)
265     {
266         var d = dot( refNormal, IncPoly.normals[i]);
267         if(d < minDot)
268         {
269             minDot = d;
270             incFace = i;
271         }
272     }
273 
274     // Almaceno los puntos de contacto
275     v[0] = vec2_add(vec2_transform(IncPoly.matWorld , IncPoly.vertices[incFace]) , IncBody.position);
276     incFace = incFace + 1 >= IncPoly.m_vertexCount ? 0 : incFace + 1;
277     v[1] = vec2_add(vec2_transform(IncPoly.matWorld , IncPoly.vertices[incFace]) , IncBody.position);
278 }

280 function clipFace( n, c, face)
281 {
282     var sp = 0;
283     var out = [face[0].Clone(), face[1].Clone()];
284 
285     // d = ax + by - c
286     var d1 = dot( n, face[0] ) - c;
287     var d2 = dot( n, face[1] ) - c;
288 
289     // estan dentras del plano? 
290     if (d1 <= 0)
291         out[sp++] = face[0].Clone();
292 
293     if (d2 <= 0)
294         out[sp++] = face[1].Clone();
295 
296     // estan en semiplanos diferentes ? 
297     if(d1 * d2 < 0)
298     {
299         // meto el punto de interseccion
300         var alpha = d1 / (d1 - d2);
301         out[sp] = vec2_add(face[0] , vec2_multiply(vec2_substract(face[1] , face[0]) , alpha));
302         ++sp;
303     }
304 
305     face[0] = out[0];
306     face[1] = out[1];
307     return sp;
308 }

310 /**
311  * Colision entre dos Poligono (convexos)
312  * implementa el algoritmo SAT
313  * @function PolygonPolygon
314  * @param {Body} a poligono a 
315  * @param {Body} b poligono b 
316  * @param {ContactSolver} contact_info informacion de contacto
317  */
318 function PolygonPolygon( a, b ,contact_info)
319 {
320 	var A = a.shape;
321 	var B = b.shape;
322 
323 	contact_info.contact_count = 0;
324 
325 	if (b.position.y > 45)
326 	    var vp = 1;
327 
328     // Busco un "separating axis" en las caras del poligono A
329 	var faceA = new Vector2(0,0);           // uso solo la x, pero no tenia forma de pasar por referencia un valor int
330 	var penetrationA = getSeparationAxis(faceA, a, b);
331     // 
332 	if (penetrationA > EPSILON)
333 		return false;
334 
335     // Busco un "separating axis" en las caras del poligono B
336 	var faceB = new Vector2(0, 0);
337 	var penetrationB = getSeparationAxis( faceB, b, a );
338 	if(penetrationB > EPSILON)
339 		return false;
340 
341 	var refIndex;
342 	var flip;       // Always point from a to b
343 	var RefBody;    // Reference
344 	var IncBody;    // Incident
345 
346     // Cual de los 2 poligonos contiene el "reference face" refFace
347 	var k_biasRelative = 0.95;
348 	var k_biasAbsolute = 0.01;
349 	if (penetrationA >= penetrationB * k_biasRelative + penetrationA * k_biasAbsolute)
350     {
351 		RefBody = a;
352 		IncBody = b;
353 		refIndex = faceA.x;
354 		flip = false;
355     }
356     else
357     {
358 		RefBody = b;
359 		IncBody = a;
360 		refIndex = faceB.x;
361 		flip = true;
362     }
363     
364 	RefPoly = RefBody.shape;        // Reference
365 	IncPoly = IncBody.shape;        // Incident
366 
367 	var incFace = [new Vector2(0,0) , new Vector2(0,0)];
368 	getIncFace( incFace, RefBody, IncBody, refIndex);
369 	var v1 = RefPoly.vertices[refIndex];
370 	var v2 = RefPoly.vertices[refIndex+1];
371 
372     // Transformo a world psace
373 	v1 = vec2_add(vec2_transform(RefPoly.matWorld , v1) , RefBody.position);
374 	v2 = vec2_add(vec2_transform(RefPoly.matWorld , v2) , RefBody.position);
375     // computo el reference 
376 	var sidePlaneNormal = vec2_substract(v2 , v1);
377 	sidePlaneNormal.normalize( );
378 
379 	var refFaceNormal = new Vector2(sidePlaneNormal.y, -sidePlaneNormal.x);
380 
381     // ecuacion de la recta
382     // ax + by = c
383     // c es la distancia al origen
384 	var refC = dot( refFaceNormal, v1 );
385 	var negSide = -dot( sidePlaneNormal, v1 );
386 	var posSide =  dot( sidePlaneNormal, v2 );
387 
388     // determino si el punto esta dentro de los limites de la arista (clipping)
389 	if (clipFace(vec2_multiply(sidePlaneNormal, -1), negSide, incFace) < 2)
390 		return false; 
391 
392 	if(clipFace(  sidePlaneNormal, posSide, incFace ) < 2)
393 		return false; 
394 
395     // Flip= indica que quedo al reves, hay que usar -dir
396 	contact_info.normal = flip ? vec2_multiply(refFaceNormal,-1) : refFaceNormal;
397 
398     // computo los puntos de contacto pp dichos
399 	var cp = 0; 
400 	var separation = dot( refFaceNormal, incFace[0] ) - refC;
401 	if(separation <= 0)
402     {
403 		contact_info.contacts[cp] = incFace[0];
404 		contact_info.penetration = -separation;
405 		++cp;
406     }
407     else
408 		contact_info.penetration = 0;
409 
410 	separation = dot( refFaceNormal, incFace[1] ) - refC;
411 	if(separation <= 0)
412     {
413 		contact_info.contacts[cp] = incFace[1];
414 		contact_info.penetration += -separation;
415 		++cp;
416 
417         // si hay mas 2 un punto de contacto, promedio la penetracion
418 		contact_info.penetration /= cp;
419     }
420 
421 	contact_info.contact_count = cp;
422 	return true;
423 }
