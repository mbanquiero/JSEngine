 10 /**
 11  * Constraint
 12  * @class Constraint Restricciones entre dos cuerpos A y B
 13  * @property {Body} A cuerpo A 
 14  * @property {Body} B cuerpo B 
 15  * @property {Vector2} ptA pivote o punto de contacto con el cuerpo A (si no esta definido usa el Centro de masas)
 16  * @property {Vector2} ptB pivote o punto de contacto con el cuerpo B (si no esta definido usa el Centro de masas)
 17  * @property {float} dist parametros del constraint: distancia
 18  * @property {float} angle parametros del constraint: angulo
 19  * @property {boolean} joint flag que indica si el constraint es del tipo articulacion
 20  * @property {boolean} exact_distance flag que indica si el constraint de distancia es exacto, o si la distancia es menor o igual
 21  * @property {boolean} angle_constraint flag que indica si el constraint del tipo restriccion de angulo en una articulacion
 22  * @property {boolean} visible flag que indica si el graficador tiene que dibujar el constraint
 23  * @constructor
 24  * @param {Body} A cuerpo A 
 25  * @param {Body} B cuerpo B 
 26  * @param {float} dist distancia
 27  */
 28 
 29 function Constraint(A,B,dist)
 30 {
 31     // cuerpos que intervienen
 32     this.A = A;
 33     this.B = B;
 34     // pivotes: x defecto el pto se toma en el centro de gravedad
 35     this.ptA = new Vector2(0,0);
 36     this.ptB = new Vector2(0,0);
 37     // parametros del constraint
 38     this.dist = dist;
 39     this.angle = 0;
 40     // propiedades
 41     this.joint = false;
 42     this.exact_distance = false;
 43     this.angle_constraint = false;
 44     this.visible = 1;
 45 };
 46 
 47 /**
 48  * Crea un constraint de distancia entre 2 cuerpos y en una posicion especifica (pivotes de contacto)
 49  * @function CreateDistanceConstraint
 50  * @param {Body} A cuerpo A 
 51  * @param {Vector2} ptA pivote o punto de contacto con el cuerpo A 
 52  * @param {Body} B cuerpo B 
 53  * @param {Vector2} ptB pivote o punto de contacto con el cuerpo B 
 54  */
 55 function CreateDistanceConstraint(A, ptA, B, ptB) {
 56     var c = new Constraint(A, B, 0);
 57     c.ptA = ptA;
 58     c.ptB = ptB;
 59     var rA = vec2_transform(A.shape.matWorld, ptA);
 60     rA.add(A.position);
 61     var rB = vec2_transform(B.shape.matWorld, ptB);
 62     rB.add(B.position);
 63     var r = vec2_substract(rA, rB);
 64     c.dist = r.length();
 65     c.joint = false;
 66     c.angle_constraint = false;
 67     c.exact_distance = true;
 68     c.visible = 1;
 69     return c;
 70 }

 73 Constraint.prototype.Solve = function()
 74 {
 75     // proyecto los puntos a world space y obtengo la distancia entre los pivotes
 76     var A = this.A;
 77     var B = this.B;
 78 
 79     var rA = vec2_transform(A.shape.matWorld , this.ptA);
 80     var rB = vec2_transform(B.shape.matWorld, this.ptB);
 81 
 82     // vector entre los puntos de contacto en world space
 83     // vr = B->position + rB - A->position - rA;
 84     var vr = vec2_substract(vec2_add(rB, B.position), vec2_add(rA, A.position));
 85 
 86     // verifico si cumple con el constraint
 87     var dreal = vr.length();
 88     if(this.exact_distance)
 89     {
 90         if(Math.abs(dreal-this.dist)<0.01)
 91             return;         // verifica el constraint, vuelvo
 92     }
 93     else
 94     {
 95         if(dreal<this.dist)
 96             return;         // verifica el constraint, vuelvo
 97     }
 98 	
 99     // no se cumple la restriccion de distancia
100     var n = vec2_multiply(vr , (1/dreal));		// direccion de A hacia B normalizada
101     var D = dreal - this.dist;			// distancia que tengo que corregir 
102     var dt = fixed_dt;			// en cierta cantidad de tiempo
103 
104     // Para ello genero una fuerza ficticia en direccion hacia el objeto B, que compense la velocidad relativa de A
105     // y lo haga dirigirse hacia B 
106     // tomo solo la parte de la velocidad de A sobre la linea de fuerza n
107     // uso la velocidad relativa entre ambos cuerpos
108     var velRel = vec2_substract(A.velocity , B.velocity);
109     var j = D/dt -dot(n,velRel);
110     j*=0.3;
111 
112     // distribuyo el impulso de forma proporcional a las masas de ambos cuerpos
113     var t;
114     if(A.inverseMass==0)
115         t = 0;
116     else
117     if(B.inverseMass==0)
118         t = 1;
119     else
120         t = A.mass / (A.mass + B.mass);
121 
122     var jA = j * t;
123     var jB = j * (1-t);
124 
125 
126     // Primero corrijo el Objeto A
127     var nA = vec2_multiply(n,jA);
128     if(A.inverseMass!=0)
129         A.velocity.add(nA);
130     if(A.inverseInertia!=0)
131         A.SetOrient( A.orient + A.inverseInertia * cross_vec(rA,nA));
132 
133 
134     var nB = vec2_multiply(n,-jB);
135     if(B.inverseMass!=0)
136         B.velocity.add(nB);
137     if(B.inverseInertia!=0)
138         B.SetOrient( B.orient + B.inverseInertia * cross_vec(rB,nB));
139 
140     if(this.joint)
141     {
142         // limito el angulo entre los 2 cuerpos, hardcodeado en pi/8
143         var limit = Math.PI/8.0;
144         var dW = A.orient-B.orient;
145         if(Math.abs(dW)>limit)
146         {
147             // corrijo un 80% del desfasaje angular
148             var desf = (Math.abs(dW)-limit) *.8;
149             if(A.orient<B.orient)
150                 desf *= -1;
151 
152             if(A.inverseInertia!=0 )
153                 A.SetOrient( A.orient - desf/2);
154             if(B.inverseInertia!=0 )
155                 B.SetOrient( B.orient + desf/2);
156 
157         }
158     }
159 }
