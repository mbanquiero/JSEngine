 25 GeoCircle.prototype.ComputeMass = function(density , body)
 26 {
 27     var r2 = this.radius * this.radius;
 28     // computo los inversos de la masa y la inercia. En la mayor parte de las ecuaciones de la dinamica de cuerpos rigidos
 29     // la masa y la inercia aparecen dividiendo, con lo cual es mas practico almacenar 1/mass y 1/inertia 
 30     // masa
 31     body.mass = Math.PI * r2 * density;
 32     if(body.mass!=0)
 33         body.inverseMass = 1 / body.mass;
 34     // Momento de inercia
 35     body.inertia = body.mass * r2;
 36     if(body.inertia!=0)
 37         body.inverseInertia = 1 / body.inertia;
 38 
 39 
 40 }

 77     if(this.drawCM)

136 GeoPolygon.prototype.CreatePoly = function(p,count)
137 {
138     // copio los vertices
139     this.m_vertexCount = count;
140     for(var i = 0; i < count; ++i)
141         this.vertices[i] = p[i].Clone();
142     // cierro la figura
143     this.vertices[count] = p[0].Clone();
144 
145     // calculo las normales
146     for(var i = 0; i < count; ++i)
147     {
148         var face = vec2_substract(this.vertices[i], this.vertices[i + 1]);
149         this.normals[i] = vec2_normal(face);
150         this.normals[i].normalize( );
151     }
152     this.normals[count] = this.normals[0].Clone();
153     this.matWorld = new Matrix2(1,0,0,1);
154 }

167 GeoPolygon.prototype.ComputeMass = function( density , body)
168 {
169     // Calcula el area  
170     // http://en.wikipedia.org/wiki/Centroid
171     // y el momento de inercia
172     // http://en.wikipedia.org/wiki/Second_moment_of_area
173     // formulas y ecuaciones 
174     // http://richardson.eng.ua.edu/Former_Courses//CE_331_fa09/Projects/A_and_I_of_Polygon.pdf
175     // surveyor-s formula para el calculo de area de un poligono
176     // http://steiner.math.nthu.edu.tw/disk5/js/cardioid/12.pdf
177 
178 
179     // Se supone que el pto N = pto 0 , ya que el poligono es cerrado
180     this.vertices[this.m_vertexCount] = this.vertices[0].Clone();
181 
182     var c = new Vector2( 0.0, 0.0 ); // centro de masa
183     var area = 0.0;
184     var Ixy = 0.0;
185 
186     for(var i = 0; i < this.m_vertexCount; ++i)
187     {
188         var p1 = this.vertices[i];
189         var p2 = this.vertices[i+1];
190         var D = cross_vec( p1, p2 );
191         var triangleArea = 0.5 * D;
192         area += triangleArea;
193         c.add( vec2_multiply(vec2_add(p1 , p2) , (triangleArea / 3.0)));
194         var xx = p1.x * p1.x + p2.x * p1.x + p2.x * p2.x;
195         var yy = p1.y * p1.y + p2.y * p1.y + p2.y * p2.y;
196         Ixy += (0.25 * D / 3.0) * (xx + yy);
197     }
198     c.multiply(1/area);
199     
200     body.mass = density * area;
201     if(body.mass!=0)
202         body.inverseMass = 1 / body.mass;
203 
204     // Momento de inercia
205     body.inertia = density * Ixy;
206     if(body.inertia!=0)
207         body.inverseInertia = 1 / body.inertia;
208 
209 	
210     // Ajusto la posicion de los vertices para que la posicion de origen (0,0) coincida con el centro de masas
211     for(var i = 0; i <= this.m_vertexCount; ++i)
212     {
213         this.vertices[i].x -= c.x;
214         this.vertices[i].y -= c.y;
215     }
216 	
217 	this.cm.x = c.x;
218 	this.cm.y = c.y;
219 	
220 }

234 {
235     var min_x = 10000;
236     var min_y = 10000;
237     var max_x = -10000;
238     var max_y = -10000;
239 
240     for (var i = 0; i <= this.m_vertexCount; ++i) {
241         var x = this.vertices[i].x;
242         var y = this.vertices[i].y;
243         if (x < min_x)
244             min_x = x;
245         if (x > max_x)
246             max_x = x;
247 
248         if (y < min_y)
249             min_y = y;
250         if (y > max_y)
251             max_y = y;
252     }
253     body.width = max_x - min_x;
254     body.height = max_y - min_y;
255 }

289 {
290     var maxProjection = -10000;
291     var rta;
292     for(var i = 0; i < this.m_vertexCount; ++i)
293     {
294         var v = this.vertices[i];
295         var projection = dot( v, dir );
296 
297         if (projection > maxProjection)

299             rta = v;
300             maxProjection = projection;
301         }
302     }

433 // Aplica una impulso al cuerpo rigido
434 // Un impulso es una fuerza que produce un cambio instantaneo en la velocidad. 
435 // Matematicamente es una fuerza de gran intensidad aplicada durante  un periodo corto de tiempo. 
436 // http://es.wikipedia.org/wiki/Delta_de_Dirac
437 
438 // Desde el punto de vista de la dinamica del cuerpo rigido, para nosotros el impulso es un cambio instantaneo en el vector
439 // de velocidad, (es decir directamente cambiamos la velocidad sin tener que pasar por la aceleracion)
440 // En la vida real los cambios de velocidad se dan apartir de una aceleracion, producto de aplicar una fuerza, segun
441 // la segunda ley de movimiento de Newton
442 // F = ma 
443 // Sin embargo, el tiempo de aplicacion de la fuerza es menor que el elapsed_time de la simulacion, con lo cual 
444 // numericamente, es preferible modificar directamente la velocidad.
445 // El impulso produce un cambio en la velocidad (traslacion) del cuerpo rigido, que es inversamente proporcional a la masa. 
446 // -------------------------------------------------------------
447 // ecuacion (1)                 F = ma
448 // -------------------------------------------------------------
449 
450 // Si el impulso se produce en alguna linea de accion que pasa por el centro de masas (CM), solo se produce una movimiento 
451 // de traslacion. Pero en cualquier otro eje de acciÃ³n, la fuerza tiene un cierto "momento" 
452 // El momento de la fuerza con respecto al eje, es el responsable de un cambio en el movimiento de rotacion. 
453 // http://es.wikipedia.org/wiki/Momento_de_fuerza
454 // En  terminos simples el momento de la fuerza es tanto mayor cuanto mas alejado ese del eje que pasa sobre el CM
455 // Por ejemplo, si queremos abrir una puerta aplicamos una fuerza sobre la manija. Si quisieramos abrir la puerta aplicando
456 // esa misma fuerza sobre el centro de la fuerza, nos costaria mucho mas.
457 // Por ultimo si quisieramos abrir la puerta aplicando la fuerza sobre la bisagra, no podriamos conseguir abrir la puerta
458 // La fuerza aplicada sobre el eje que pasa por el CM no tiene momento (el momento es nulo). 
459 // Si el momento de la fuerza es cero, la fuerza no produce rotacion.
460 
461 // Para computar el momento de la fuerza, tambien llamado torque, se usa el producto vectorial del vector posicion del
462 // punto donde se aplica la fuerza,por el vector fuerza 
463 // T = r x F        T = torque, r = vector posicion, F = fuerza
464 // 
465 
466 // En nuestro caso el punto de aplicacion es contactPoint y el vector fuerza es impuse
467 // Torque = cross_vec(contactPoint, impulse)
468 // En 2 dimensiones, el producto cruz no es tan "intuitivo" como en 3d dimensiones. Es decir, no hay una forma 
469 // standard de computar un cross product en 2d. 
470 // En nuestro caso, lo que se precisa es la magnitud de lo que seria el cross product 3d si la coordenada Z fuerse cero
471 // Esa magnitud es la que va a producir un cambio en al velocidad de rotacion, o velocidad angular. 
472 
473 // Ahora bien, no todos los torques producen la misma rotacion en los cuerpos rigidos, dependen de la resistencia que tenga
474 // el cuerpo rigido de rotar sobre el eje particular. 
475 // Ese concepto se llama momento de inercia, y es funciona de la misma forma que la masa para la aceleracion lineal.
476 
477 // Cuando queremos mover un objeto con mas masa tenemos que aplicar mas fuerza. La masa se puede ver como la resistencia
478 // que impone el cuerpo al movimiento lineal. Y como siempre esta descripta por la ecuacion de newton F = ma 
479 
480 // De la misma forma, cuando hablamos de movimientos de rotacion, el analogo a la fuerza es el "momento de la fuerza", o torque
481 // y el analogo de la masa es el momento de inercia. El momento de inercia = I, depende no solo de la masa, (cuerpos mas masivos
482 // son mas dificiles de hacer girar que cuerpos mas livianos), si no tambien ademas de la distribuicion de la masa
483 // con respecto al eje sobre el que se quiere hacer girar. 
484 // http://es.wikipedia.org/wiki/Momento_de_inercia
485 
486 // La ecuacion que describe estos 2 conceptos es 
487 // -------------------------------------------------------------
488 // ecuacion (2)
489 // T = I.dw/dt          T = torque, I = momento de inercia, dw/dt = aceleracion angular
490 // -------------------------------------------------------------
491 // http://es.wikipedia.org/wiki/Aceleraci%C3%B3n_angular
492 // -------------------------------------------------------------
493 
494 
495 Body.prototype.ApplyImpulse = function (impulse, contactPoint)
496 {
497     if (this.inverseMass == 0) {
498         if (this.mass == 0)
499             return;     // es un cuerpo estatico
500             /*
501         else {
502             // si inverseMass = 0 pero mass >0 es porque era un cuerpo dinamico, que momentaneamente estaba 
503             // en equilibrio ( se comportaba como un cuerpo estatico)
504             // eso era valido mientras la unica fuerza que actuaba sobre el cuerpo es la gravedad
505             // Si le aplicamos un impulso, es porque hay otra fuerza, se rompio el equilibrio, y se vuelve a comportar como dinamico
506             this.inverseMass = 1 / this.mass;
507         }*/
508     }
509 
510 
511     var mJ = impulse.length();
512 
513     if (mJ > this.maxImpulse)
514         this.maxImpulse = mJ;
515 
516     if (this.inverseMass != 0 && this.maxForce != 0 && mJ > this.maxForce)
517     {
518         // el cuerpo revento, lo saco de la simulacion
519         this.deleted = true;
520         return;
521     }
522 
523 
524     // ecuacion (1)
525     // F = ma
526     this.velocity.add(vec2_multiply(impulse, this.inverseMass));
527 
528     // ecuacion (2)
529     // T = I.dw/dt          T = torque, I = momento de inercia, dw/dt = aceleracion angular
530     this.angularVelocity += this.inverseInertia * cross_vec(contactPoint, impulse);
531 
532 }

584 //    F = mA

594 // Para integrar conviene hacerlo en 2 etapas, avanzando el tiempo por la mitad. La explicacion se puede encontrar aca
595 // http://www.niksula.hut.fi/~hkankaan/Homepages/gravity.html

599 Body.prototype.IntegrateForces = function( dt)
600 {
601     if(this.inverseMass == 0)
602         return;
603 
604     //	velocity += (force * inverseMass + gravity*gravity_factor) * dt;
605 
606     // aceleracion que producen las fuerzas externas
607     var acel = vec2_multiply(this.force, this.inverseMass);
608     // aceleracion de la gravedad
609     var acel_grav = vec2_multiply(gravity, this.gravity_factor);
610     // aceleracion neta
611     var acel_neta = vec2_add(acel, acel_grav);
612     // integro la aceleracion lineal
613     this.velocity.add(vec2_multiply(acel_neta, dt));
614     // integro la aceleracion angular
615     this.angularVelocity += this.torque * this.inverseInertia * dt;
616 
617 }
618 
619 Body.prototype.IntegrateVelocity = function(dt)
620 {
621     if(this.inverseMass == 0)
622         return;
623 
624     this.position.add( vec2_multiply(this.velocity ,  dt));
625     this.orient += this.angularVelocity * dt;
626     this.orient = fmod(this.orient, 2.0 * Math.PI);
627     this.SetOrient(this.orient);
628 
629 }

631 Body.prototype.Render = function()
632 {
633 
634     if (!this.visible)
635         return;
636 
637     if (this.img != null) {
638         // tiene una imagen asociada, dibuja atraves de la imagen
639 
640         var x = ox + this.position.x*ex;
641         var y = oy + this.position.y*ey;
642         var width = this.width * ex;
643         var height = this.height * ey;
644 
645         ctx.translate(x, y);
646         ctx.rotate(this.orient);
647         ctx.drawImage(this.img, -width / 2, -height / 2, width, height);
648         ctx.rotate(-this.orient);
649         ctx.translate(-x, -y);
650     }
651     else {
652         // dibujo a traves de la interface de geometria asociada al cuerpo
653         this.shape.Render(this);
654     }

670         ctx.closePath();
671         ctx.fill();
672         alfa += da;
