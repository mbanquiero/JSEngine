 43 /**
 44  * Genero informacion del contacto entre todos los cuerpos: Narrow Phase
 45  * @method Update
 46  * @param {float} elapsed_time tiempo transcurrido
 47 */
 48 World.prototype.Update = function(elapsed_time)
 49 {
 50 
 51     // Genero informacion del contacto entre todos los cuerpos: Narrow Phase
 52     this.m_contactCount = 0;
 53     for(var i = 0; i < this.m_bodyCount-1; ++i)
 54     {
 55         var A = this.bodies[i];
 56 
 57         for(var j = i + 1; j < this.m_bodyCount; ++j)
 58         {
 59             var B = this.bodies[j];
 60             // si ambos cuerpos son estaticos, no pueden colisionar
 61             if(A.inverseMass == 0 && B.inverseMass== 0)
 62                 continue;
 63 
 64             // si Ambos cuerpos forman un joint tampoco pueden colisionar, aunque si lo pueden hacer en la vida real 
 65             // el solver no es estable con este tipo de configuracion 
 66             var C = new ContactSolver();
 67             if(C.BodyCollide(elapsed_time , A,B))
 68             {
 69                 this.contacts[this.m_contactCount++] = C;
 70             }
 71         }
 72     }
 73 
 74     // Integrar fuerzas primer parte
 75     for(var i = 0; i < this.m_bodyCount; ++i)
 76         this.bodies[i].IntegrateForces( elapsed_time/2.);
 77 
 78     // Precalculo la info de contactos
 79     for(var i=0;i<this.m_contactCount;++i)
 80         this.contacts[i].PreCalc();
 81 
 82     // Fase de Collision Response: Impulse Resolution
 83     var cant_iteracciones = 100;
 84     for(var j = 0; j < cant_iteracciones; ++j)
 85     for(var i = 0; i < this.m_contactCount; ++i)
 86         this.contacts[i].ApplyImpulse();
 87 		
 88     // Integrar velocidades
 89     for(var i = 0; i < this.m_bodyCount; ++i)
 90         this.bodies[i].IntegrateVelocity(  elapsed_time );
 91 
 92     // Integrar fuerzas segunda parte
 93     for(var i = 0; i < this.m_bodyCount; ++i)
 94         this.bodies[i].IntegrateForces( elapsed_time/2.);
 95 
 96     // Corregir posiciones (eliminar penetraciones)
 97     for(var i = 0; i < this.m_contactCount; ++i)
 98         this.contacts[i].positionCorrection( );
 99 
100     // Limpiar todas las fuerzas, pues ya fueron integradas
101     for(var i = 0; i < this.m_bodyCount; ++i)
102     {
103         this.bodies[i].force.Set(0,0);
104         this.bodies[i].torque = 0;
105     }
106 
107     // Solve constrains
108     for(var j = 0; j < 3; ++j)
109         for(var i = 0; i < this.m_constraintCount; ++i)
110             this.constraints[i].Solve();
111 
112 
113     // saco de la simulacion los objetos borados
114     for (var i = 0; i < this.m_bodyCount; ++i) {
115         var A = this.bodies[i];
116         if (A.deleted) {
117             A.position.Set(0, 0);
118             A.SetStatic();
119         }
120     }
121 
122 
123     // avanzo el tiempo global
124     this.time += elapsed_time;
125 
126 }

165 /**
166  * @method Render
167 */
168 World.prototype.Render = function()
169 {
170     // Dibujo los cuerpos
171     ctx.font = "12px Arial";
172     for(var i = 0; i < this.m_bodyCount; ++i)
173     {
174         var A = this.bodies[i];
175         A.Render();
176 
177         /*
178         // debug max impulse
179         if (A.inverseMass != 0 && A.maxImpulse!=0)
180         {
181             //var W = _world.ComputeWeight(A);
182             //ctx.fillText("" + W.toFixed(1), ox + A.position.x * ex, oy + A.position.y * ey-20);
183             ctx.fillText("" + A.maxImpulse.toFixed(1), ox + A.position.x * ex, oy + A.position.y * ey-20);
184         }
185         */
186     }
187 
188     // Dibujo los contactos
189     ctx.fillStyle = '#FFFFFF';
190     ctx.strokeStyle = '#3030FF';
191     for(var i = 0; i < this.m_contactCount; ++i)
192     {
193         for(var j = 0; j < this.contacts[i].contact_count; ++j)
194         {
195             var c = this.contacts[i].contacts[j];
196             var n = this.contacts[i].normal;
197 
198             // dibujo un circulo en la posicion de contacto
199             ctx.beginPath();
200             ctx.arc(ox + c.x*ex, oy + c.y*ey, 5, 0, Math.PI*2,true); 
201             ctx.closePath();
202             ctx.fill();
203             ctx.stroke();
204 
205             // dibujo una linea indicando la direccion normal
206             ctx.beginPath();
207             ctx.moveTo(ox + c.x*ex, oy + c.y*ey);
208             ctx.lineTo(ox + c.x *ex + n.x*20, oy + c.y*ey+n.y*20);
209             ctx.stroke();
210         }
211     }
212 
213 
214     // dibujo los constraints
215     ctx.strokeStyle = '#F0F0FF';
216     for(var i = 0; i < this.m_constraintCount; ++i)
217     {
218         switch(this.constraints[i].visible)
219         {
220             case 1:
221                 // dibujo una linea llena fina 
222                 ctx.beginPath();
223                 var c = vec2_transform(this.constraints[i].A.shape.matWorld, this.constraints[i].ptA);
224                 c.add(this.constraints[i].A.position);
225                 ctx.moveTo(ox + c.x*ex ,oy + c.y*ey);
226 
227                 c = vec2_transform(this.constraints[i].B.shape.matWorld , this.constraints[i].ptB);
228                 c.add(this.constraints[i].B.position);
229                 ctx.lineTo(ox + c.x * ex, oy + c.y * ey);
230                 ctx.stroke();
231                 break;
232 
233             case 2:
234                 // dibujo una linea gruesa (usualmente son joints y quiero dibujar el hueso)
235                 var p0 = vec2_transform(this.constraints[i].A.shape.matWorld, this.constraints[i].ptA);
236                 p0.add(this.constraints[i].A.position);
237                 var p1 = vec2_transform(this.constraints[i].B.shape.matWorld, this.constraints[i].ptB);
238                 p1.add(this.constraints[i].B.position);
239                 var dir_v = vec2_substract(p1 , p0);
240                 dir_v.normalize();
241                 var dir_w = vec2_normal(dir_v);
242                 var rA = this.constraints[i].A.shape.radius;
243                 var rB = this.constraints[i].B.shape.radius;
244                 var Q = [];
245                 Q[0] = vec2_add ( p0  , vec2_multiply(dir_w , rA));
246                 Q[1] = vec2_add ( p1  , vec2_multiply(dir_w , rB));
247                 Q[2] = vec2_substract ( p1  , vec2_multiply(dir_w , rB));
248                 Q[3] = vec2_substract(p0, vec2_multiply(dir_w, rA));
249 
250                 ctx.fillStyle = '#f0f0f0';
251                 ctx.beginPath();
252                 ctx.moveTo(ox + Q[0].x * ex, oy + Q[0].y * ey);
253                 for(var t=1;t<4;++t)
254                     ctx.lineTo(ox + Q[t].x * ex, oy + Q[t].y * ey);
255                 ctx.closePath();
256                 ctx.fill();
257                 ctx.stroke();
258                 break;
259         }
260     }
261 }

388 /**
389  * interaccion c/ usuario: devuelve si el pto esta cercano al centro de gravedad de un cuerpo
390  * @method PointNearBodyCM
391  * @param {Vector2} pt posicion del mouse 
392  * @param {float} dE -delta epsilon de contacto
393 */
394 World.prototype.PointNearBodyCM = function(pt, dE)
395 {
396 	var rta = -1;
397     var min_dist = dE*dE;
398     for(var i = 0;i<this.m_bodyCount;++i)
399     if(this.bodies[i].inverseMass!=0)		// solo se aplica a objetos dinamicos
400     {
401         var dist = vec2_substract(pt, this.bodies[i].position).lengthSqr();
402         if(dist<min_dist)
403         {
404             rta = i;
405             min_dist = dist;
406         }
407     }
408     return rta;
409 }
