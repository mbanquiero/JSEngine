  5 <P> Collision response (impulse resolution)</P>
  6 <P> Los motores de fisicas usualmente tienen 2 etapas bien definidas</P>
  7 <P> * Etapa de Collision Detection: el objetivo de esta etapa es determinar si 2 o mas cuerpos entran en colision. </P>
  8 <P> PAra ello se aplican una serie de algoritmos geometricos bastante estudiados. </P>
  9 <P> Esta etapa a su vez se puede dividir en 2 fases :</P>
 10 <P>              - Broad Phase. </P>
 11 <P>	Se trata de rapidamente descartar aquellos pares de cuerpos que por estar lo suficienemente
 12                               alejados no pueden entrar en colision. Se suelen usar bounding box y otro tipo de simplificaciones
 13                               y distintas estructuras jerarquicas. La idea es que rapidamente se puedan detectar pares de 
 14                               candidatos que pueden entrar en colision.</P>
 15 							  
 16 <P>              - Narrow Phase. </P>
 17 <P>Con el resultado de pares "posibles" de la fase anterior, se determinar exactamente si 
 18                               esos "candidatos" a colision, realmente estaban colisionando, y en caso afirmativo
 19                               colectar informacion especifica de la colision, que luego va a servir en etapas posteriores
 20                               como el punto exacto de la colision, la normal, y la penetracion.</P>
 21 							  
 22 <P> * Etapa de Collision Response : una vez dectectada la collision, el motor tiene que tomar alguna acción para evitar
 23  que los cuerpos que collisionaron siguen en esa situacion. En nuestro caso se resuelve aplicando un impulso de tal forma
 24  que los cuerpos se alejen en la direccion que chocaron,  y asi evitar que sigan en colision = Impulse Resolution.</P>
 25 <P> A continuacion se codifican los algortimos de Impulse Resolution.</P>
 26 <P> Impulse Resolution es una de las tantas formas estandard de los motores de fisica de resolver la colision entre cuerpos
 27  La idea es aplicar un impulso (cambio instantaneo en la velocidad) a los cuerpos que entran en colision, de tal manera
 28  que tienan a separarse. </P>
 29 <P>Los algoritmos que detectan la colision tambien informan la normal de la colision, el punto de contacto y la penetracion</P>
 30 */

 33 /**
 34  * @class ContactSolver contactos entre 2 cuerpos
 35  * @property {Body} A cuerpo A 
 36  * @property {Body} B cuerpo B 
 37  * @property {float} penetracion  Cuanto penetra el Cuerpo A dentro del B durante el contacto
 38  * @property {Vector2} normal Vector de A to B
 39  * @property {Vector2[]} contacts Array de Puntos de contacto. 
 40  * @property {int} contact_count Cantidad de puntos de contacto
 41  * @property {float} dt elapsed time
 42  * @property {float} e Coeficiente de restitution
 43  * @property {float} df Coeficiente de friccion dinamica
 44  * @property {float} sf Coeficiente de friccion estatica
 45  * @property {boolean} resting_contact flag que indica que es un contacto tipo "resting", por ejemplo un libro apoyado sobre una mesa
 46  * @constructor
 47  */
 48 
 49 function ContactSolver()
 50 {
 51     this.A = null;                      // cuerpo A
 52     this.B = null;                      // cuerpo B
 53     this.penetration = 0;			    // Cuanto penetra el Cuerpo A dentro del B durante el contacto
 54     this.normal = new Vector2(0,0);		// Vector de A to B
 55     this.contacts = [];		            // Puntos de contacto. 
 56     this.contact_count = 0;			    // Cantidad de puntos de contacto
 57     this.dt = 0;					    // elapsed time
 58     this.e = 0;					        // Coeficiente de restitution
 59     this.df = 0;					    // Coeficiente de friccion dinamica
 60     this.sf = 0;					    // Coeficiente de friccion estatica
 61     this.resting_contact = false;
 62 }

 66 /**
 67  * 
 68  * @method BodyCollide Determina si hay contacto entre el cuerpo A y el Cuerpo B
 69  * @param {float} elapsed_time tiempo transcurrido
 70  * @param {Body} A body A
 71  * @param {Body} B body B
 72 */
 73 
 74 ContactSolver.prototype.BodyCollide = function(elapsed_time, pA , pB)
 75 {
 76     this.dt = elapsed_time;
 77     this.A = pA;
 78     this.B = pB;
 79     this.contact_count = 0;
 80 
 81     if(pA.shape.GetType()==GPE_CIRCLE && pB.shape.GetType()==GPE_CIRCLE)
 82         CircleCircle(this.A, this.B, this);
 83     else
 84     if(pA.shape.GetType()==GPE_CIRCLE && pB.shape.GetType()==GPE_POLYGON)
 85         CirclePolygon(this.A, this.B, this);
 86     else
 87     if(pA.shape.GetType()==GPE_POLYGON && pB.shape.GetType()==GPE_CIRCLE)
 88     {
 89         this.A = pB;
 90         this.B = pA;
 91         CirclePolygon(this.A, this.B, this);
 92     }
 93     else
 94     if (pA.shape.GetType() == GPE_POLYGON && pB.shape.GetType() == GPE_POLYGON)
 95         PolygonPolygon(this.A, this.B, this);
 96     
 97     return this.contact_count > 0 ?  true : false;
 98 }

101 // El cuerpo A y el B entraron en colision, precomputa ciertos parametros para resolver luego el impulso
102 // La idea es modificar la velocidad de los cuerpos para que se alejan en direccion a la normal donde hicieron contacto
103 // De esta forma se evitara que sigan colisionando
104 ContactSolver.prototype.PreCalc = function ()
105 {
106     // El coeficiente de restitucion indica que tan elastico es el choque, o cuanta energia se pierde por el impacto
107     // usualmente se toma el minimo entre los 2 cuerpos que entran en colision,
108 	// pero yo voy a tomar el maximo
109     this.e = Math.max( this.A.restitution, this.B.restitution );
110 
111     // Computo los coeficientes de friccion (Ley de Coulumb)
112     this.sf = Math.sqrt( this.A.staticFriction * this.B.staticFriction );
113     this.df = Math.sqrt( this.A.dynamicFriction * this.B.dynamicFriction );
114 
115     for(var i = 0; i < this.contact_count; ++i)
116     {
117 
118         // la velocidad de un punto en el cuerpo rigido se puede computar como la velocidad de su centro de masas
119         // producto del movimiento lineal del cuerpo, mas la vecocidad tangencial, producto de su rotacion
120         // Velocidad tangencial depende de la distancia al centro de masas
121         // Vt = w x r 
122 
123         // Computo los radios desde el pto de contacto hasta el centro de masa 
124         var ra = vec2_substract(this.contacts[i], this.A.position);
125         var rb = vec2_substract(this.contacts[i], this.B.position);
126 
127         // computo la velocidad de los puntos de contacto 
128         var va = vec2_add(this.A.velocity, cross(this.A.angularVelocity, ra));
129         var vb = vec2_add(this.B.velocity, cross(this.B.angularVelocity, rb));
130 
131         // Computa la velocidad relativa. Es como si uno de los cuerpos estuviera fijo, y el otro se mueve con
132         // la velocidad total del sistema. Es un cambio de referencia. 
133         var rv = vec2_substract(vb , va);
134 
135         // Verifico si se trata de "resting contact" 
136         // Se llama resting contact a los contactos en equilibrio estatico, por ejemplo un libro sobre la mesa
137         // la idea es que si la unica fuerza que acelera el objeto es la de gravedad
138         // la colision se tiene que hacer sin ninguna restitucion, para evitar que el cuerpo rebote indefinidamente
139         if (rv.lengthSqr() < gravity.lengthSqr() * this.dt * this.dt + EPSILON)
140         {
141             this.e = 0;
142         }
143             
144     }
145 
146     /*
147     // parche para saber si es un resting contact estatico.
148     // si el cuerpo esta en equilibrio estatico
149     if(this.contact_count>=2 && this.e==0 && (this.A.inverseMass==0 || this.B.inverseMass==0))
150     {
151         this.resting_contact = true;
152     }
153     */
154 }

156 // Resolver el impulso 
157 ContactSolver.prototype.ApplyImpulse = function()
158 {
159     // si ambos objetos son esticos, corrijo la posicion directamente
160     if(this.A.inverseMass  + this.B.inverseMass < EPSILON)
161     {
162         this.staticBodyCorrection();
163         return;
164     }
165 
166     if (this.resting_contact)
167     {
168         this.staticBodyCorrection();
169         return;
170     }
171 
172 
173     for(var i = 0; i < this.contact_count; ++i)
174     {
175         // la velocidad de un punto en el cuerpo rigido se puede computar como la velocidad de su centro de masas
176         // producto del movimiento lineal del cuerpo, mas la vecocidad tangencial, producto de su rotacion
177         // Velocidad tangencial depende de la distancia al centro de masas
178         // Vt = w x r 
179 
180         // Computo los radios desde el pto de contacto hasta el centro de masa 
181         var ra = vec2_substract(this.contacts[i], this.A.position);
182         var rb = vec2_substract(this.contacts[i], this.B.position);
183 
184         // la velocidad de un punto en el cuerpo rigido se puede computar como la velocidad de su centro de masas
185         // producto del movimiento lineal del cuerpo, mas la vecocidad tangencial, producto de su rotacion
186         var va = vec2_add(this.A.velocity, cross(this.A.angularVelocity, ra));
187         var vb = vec2_add(this.B.velocity, cross(this.B.angularVelocity, rb));
188 
189         // Computa la velocidad relativa. Es como si uno de los cuerpos estuviera fijo, y el otro se mueve con
190         // la velocidad total del sistema. Es un cambio de referencia. 
191         var rv = vec2_substract(vb , va);
192 
193         // se llama velocidad del contacto al componente de la velocidad sobre la normal del mismo. 
194         // Es la parte de la misma que interviene en la restitucion.
195         var contactVel = dot( rv, this.normal );
196 
197         // Luego, si es velocidad es positiva los objetos ya se estan separando, no tengo que hacer nada
198         if(contactVel > 0)
199             return;
200 
201         // Para corregir la orientacion del cuerpo, preciso trabajar con el momento de la fuerza con respecto al punto
202         // Eso indica que parte de la fuerza produce una acelaracion angular, o interviene en la rotacion.,
203         var raCrossN = cross_vec( ra, this.normal );				// momento de A
204         var rbCrossN = cross_vec( rb, this.normal );				// momento de B
205         var invMassSum = this.A.inverseMass + this.B.inverseMass 
206 			+ raCrossN*raCrossN * this.A.inverseInertia + rbCrossN*rbCrossN * this.B.inverseInertia;
207 
208         // Ecuacion para el impuslo escalar j
209         // notar que el impuso es invesamente proporcional a la masa
210         // ej. Si un camion choca contra una bicicleta, la bicicleta sale disperando y el cambio apenas se mueve
211         var j = -(1 + this.e) * contactVel;
212         j /= invMassSum;
213         j /= this.contact_count;
214 
215         // Aplico el impulso pp dicho
216         this.A.ApplyImpulse(vec2_multiply(this.normal, -j), ra);
217         this.B.ApplyImpulse(vec2_multiply(this.normal, j), rb);
218 
219         if (this.resting_contact) {
220             this.staticBodyCorrection();
221         }
222         
223         if (!this.resting_contact) {
224 
225             // Computo la fuerz de friccion, la fuerza es tangente la direccion de colision, es decir la direccion de colision
226             // es la normal, y la friccion es la tangente en el choque. 
227             // vuelvo a computar la velocidad relativa (luego de haber corregido las mismas con los impulsos que aplicamos )
228             va = vec2_add(this.A.velocity, cross(this.A.angularVelocity, ra));
229             vb = vec2_add(this.B.velocity, cross(this.B.angularVelocity, rb));
230             rv = vec2_substract(vb, va);
231             var t = vec2_substract(rv, vec2_multiply(this.normal, dot(rv, this.normal)));
232             if (t.lengthSqr() == 0)
233                 return;
234 
235             t.normalize();
236             // j tangent magnitude
237             var jt = -dot(rv, t);
238             jt /= invMassSum;
239             jt /= this.contact_count;
240 
241 
242             // si la frccion es muy pequeña no la tengo en cuenta para no introducir inestabilidad en el sistmea
243             if (Math.abs(jt) < 0.01)
244                 return;
245 
246             // Ley de Coulumb's para la friccion
247             // hay 2 tipos de friccion, 
248             // estatica, es la la fuerza de resitencia que impone un objeto estatico a ser movido. 
249             // dinamica, idem cuando el cuerpo se esta movimiendo
250             // por ejemplo un libro sobre la mesa esta quito y si quiero moverlo, la friccion que tiene impide que el libro
251             // se mueva es la friccion estatica. Para mover el libro tengo que hacer una fuerza superior al la de friccion estatica
252             // asi el libro se empieza a mover. A partir de ese momento, el libro ya esta en movimiento, y comienza a actuar 
253             // la friccion dinamica. 
254             var tangentImpulse = t;
255             if (Math.abs(jt) < j * this.sf)
256                 tangentImpulse.multiply(jt);
257             else
258                 tangentImpulse.multiply(-j * this.df);
259 
260             // Aplico el impulso de la friccion
261             this.A.ApplyImpulse(vec2_multiply(tangentImpulse, -1), ra);
262             this.B.ApplyImpulse(tangentImpulse, rb);
263         }
264         // paso al siguiente contacto
265     }
266 
267 }

269 // Parche para corregir la penetration
270 ContactSolver.prototype.positionCorrection = function ()
271 {
272     var k_slop = 0.05; // Penetration allowance
273     var percent = 0.4; // Penetration percentage to correct
274     var K = (Math.max(this.penetration - k_slop, 0) / (this.A.inverseMass + this.B.inverseMass)) * percent;
275     var correction = vec2_multiply(this.normal, K);
276     this.A.position.substract(vec2_multiply(correction , this.A.inverseMass));
277     this.B.position.add(vec2_multiply(correction , this.B.inverseMass));
278 }

281 // Parche para corregir los objetos estaticos
282 ContactSolver.prototype.staticBodyCorrection = function()					
283 {
284     // simplemente elimino la velocidad.
285     this.A.velocity.Set(0,0);
286     this.B.velocity.Set(0,0);
287 }
