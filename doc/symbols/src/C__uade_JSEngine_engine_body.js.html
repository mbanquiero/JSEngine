 36 /**
 37  * Computo los inversos de la masa y la inercia. En la mayor parte de las ecuaciones de la dinamica de cuerpos rigidos
 38  * La masa y la inercia aparecen dividiendo, con lo cual es mas practico almacenar 1/mass y 1/inertia 
 39  * @method ComputeMass
 40  * @param {float} density densidad 
 41  * @param {Body} body rigid body asociado
 42 */
 43 GeoCircle.prototype.ComputeMass = function(density , body)
 44 {
 45     var r2 = this.radius * this.radius;
 46     body.mass = Math.PI * r2 * density;
 47     if(body.mass!=0)
 48         body.inverseMass = 1 / body.mass;
 49     // Momento de inercia
 50     body.inertia = body.mass * r2;
 51     if(body.inertia!=0)
 52         body.inverseInertia = 1 / body.inertia;
 53 
 54 
 55 }

 85     var pos = body.position;
 86     var r = this.radius;
 87     // dibujo un circulo en la posicion del centro de masas del cuerpo
 88     ctx.fillStyle = body.color;
 89     ctx.strokeStyle = '#ff00ff';
 90     ctx.beginPath();
 91     ctx.arc(ox + pos.x*ex, oy + pos.y*ey, r*ex, 0, Math.PI*2,true); 
 92     ctx.closePath();
 93     ctx.fill();
 94     ctx.stroke();
 95 
 96     // dibujo una linea para captar la orientacion del circulo
 97     if (this.drawOrient) {
 98         ctx.beginPath();
 99         ctx.strokeStyle = '#ff0000';
100         ctx.moveTo(ox + pos.x * ex, oy + pos.y * ey);
101         var x = pos.x + Math.cos(body.orient) * r;
102         var y = pos.y + Math.sin(body.orient) * r;
103         ctx.lineTo(ox + x * ex, oy + y * ey);
104         ctx.closePath();
105         ctx.stroke();
106     }
107     // dibujo el centro de gravedad
108     if(this.drawCM)
109         drawCenterOfMass(pos.x, pos.y, 0.5, body.orient);

166  * @param {Vector2} p1 Punto1

179     this.normals[0].normalize();
180     this.normals[1].normalize();
181     this.normals[2].normalize();
182     this.matWorld = new Matrix2(1,0,0,1);
183     this.Close();

186 /**
187  * GeoPolygon
188  * Crea un Convex HULL a partir de una nube de puntos
189  * @method
190  * @param {Vector2[]} p Array de puntos
191  * @param {int} count Cantidad de puntos del Array
192  */
193 GeoPolygon.prototype.CreatePoly = function(p,count)
194 {
195     // copio los vertices
196     this.m_vertexCount = count;
197     for(var i = 0; i < count; ++i)
198         this.vertices[i] = p[i].Clone();
199     // cierro la figura
200     this.vertices[count] = p[0].Clone();
201 
202     // calculo las normales
203     for(var i = 0; i < count; ++i)
204     {
205         var face = vec2_substract(this.vertices[i], this.vertices[i + 1]);
206         this.normals[i] = vec2_normal(face);
207         this.normals[i].normalize( );
208     }
209     this.normals[count] = this.normals[0].Clone();
210     this.matWorld = new Matrix2(1,0,0,1);
211 }

233 /**
234  <P> Computo los inversos de la masa y la inercia. En la mayor parte de las ecuaciones de la dinamica de cuerpos rigidos
235   La masa y la inercia aparecen dividiendo, con lo cual es mas practico almacenar 1/mass y 1/inertia </P>
236  <P>  Calcula el area  </P> 
237  <P>  http://en.wikipedia.org/wiki/Centroid</P> 
238  <P>  y el momento de inercia</P> 
239  <P>  http://en.wikipedia.org/wiki/Second_moment_of_area</P> 
240  <P>  formulas y ecuaciones </P> 
241  <P>  http://richardson.eng.ua.edu/Former_Courses//CE_331_fa09/Projects/A_and_I_of_Polygon.pdf</P> 
242  <P> surveyor-s formula para el calculo de area de un poligono</P> 
243  <P>  http://steiner.math.nthu.edu.tw/disk5/js/cardioid/12.pdf</P> 
244  *
245  * @method ComputeMass
246  * @param {float} density densidad 
247  * @param {Body} body rigid body asociado
248 */
249 
250 GeoPolygon.prototype.ComputeMass = function( density , body)
251 {
252 
253 
254     // Se supone que el pto N = pto 0 , ya que el poligono es cerrado
255     this.vertices[this.m_vertexCount] = this.vertices[0].Clone();
256 
257     var c = new Vector2( 0.0, 0.0 ); // centro de masa
258     var area = 0.0;
259     var Ixy = 0.0;
260 
261     for(var i = 0; i < this.m_vertexCount; ++i)
262     {
263         var p1 = this.vertices[i];
264         var p2 = this.vertices[i+1];
265         var D = cross_vec( p1, p2 );
266         var triangleArea = 0.5 * D;
267         area += triangleArea;
268         c.add( vec2_multiply(vec2_add(p1 , p2) , (triangleArea / 3.0)));
269         var xx = p1.x * p1.x + p2.x * p1.x + p2.x * p2.x;
270         var yy = p1.y * p1.y + p2.y * p1.y + p2.y * p2.y;
271         Ixy += (0.25 * D / 3.0) * (xx + yy);
272     }
273     c.multiply(1/area);
274     
275     body.mass = density * area;
276     if(body.mass!=0)
277         body.inverseMass = 1 / body.mass;
278 
279     // Momento de inercia
280     body.inertia = density * Ixy;
281     if(body.inertia!=0)
282         body.inverseInertia = 1 / body.inertia;
283 
284 	
285     // Ajusto la posicion de los vertices para que la posicion de origen (0,0) coincida con el centro de masas
286     for(var i = 0; i <= this.m_vertexCount; ++i)
287     {
288         this.vertices[i].x -= c.x;
289         this.vertices[i].y -= c.y;
290     }
291 	
292 	this.cm.x = c.x;
293 	this.cm.y = c.y;
294 	
295 }

312 /**
313  * Computo el bounding rect del cuerpo 
314  * @method ComputeSize
315  * @param {Body} body rigid body asociado
316 */
317 
318 GeoPolygon.prototype.ComputeSize = function (body)
319 {
320     var min_x = 10000;
321     var min_y = 10000;
322     var max_x = -10000;
323     var max_y = -10000;
324 
325     for (var i = 0; i <= this.m_vertexCount; ++i) {
326         var x = this.vertices[i].x;
327         var y = this.vertices[i].y;
328         if (x < min_x)
329             min_x = x;
330         if (x > max_x)
331             max_x = x;
332 
333         if (y < min_y)
334             min_y = y;
335         if (y > max_y)
336             max_y = y;
337     }
338     body.width = max_x - min_x;
339     body.height = max_y - min_y;
340 }

380 // helper para algortimo de colision convexo - convexo 
381 // devuele la el vertice que esta mas lejos en distancia proyectada en direccion dir
382 GeoPolygon.prototype.getMaxProjection = function (dir)
383 {
384     var maxProjection = -10000;
385     var rta;
386     for(var i = 0; i < this.m_vertexCount; ++i)
387     {
388         var v = this.vertices[i];
389         var projection = dot( v, dir );
390 
391         if (projection > maxProjection)
392         {
393             rta = v;
394             maxProjection = projection;
395         }
396     }
397     return rta;
398 }

551 /**
552  * Interface para la creacion de cuerpos rigidos: Crea un poligono convexo a partir de una nube de puntos
553  * generando un convex hull.
554  * El solver de collisiones solo soporta poligonos convexos, (limitacion del algoritmo SAT)
555  * Ademas, por la convencion para computar normales, tiene que estar dibujado en sentido horario.
556  * Como los puntos pueden ser una nube de puntos cualquieras, usualmente generado al azar, 
557  * para asegurarse que el poligono es convexo, directamente, partimos de la nube de puntos y creamos un convex hull
558  * De eso se encarga la funcion CreatePoly
559  * @method CreatePoly
560  * @param {Vector2[]} pt Array de puntos
561  * @param {int} cant_pt cantidad de puntos
562  * @param {float} p_density densidad
563 */
564 
565 Body.prototype.CreatePoly= function(pt,cant_pt, p_density)
566 {
567     // genero un convex hull en sentido horario, a partir de la nube de puntos
568     var vertices = [];
569     var vertex_count = convexHull(pt, cant_pt, vertices);
570 
571     // creo el shape poligono, ya asegurado que es convexo y en sentido horario
572     var poly_shape = new GeoPolygon();
573     poly_shape.CreatePoly(vertices, vertex_count);
574     this.Create(poly_shape, p_density);
575 	
576 	// muevo la posicion del poligono a su centro de gravedad
577     this.position.Set(poly_shape.cm.x, poly_shape.cm.y);

596 /**
597 <P> Aplica una impulso al cuerpo rigido</P> 
598 <P> Un impulso es una fuerza que produce un cambio instantaneo en la velocidad. 
599 	Matematicamente es una fuerza de gran intensidad aplicada durante  un periodo corto de tiempo. </P> 
600 
601 <P> http:*es.wikipedia.org/wiki/Delta_de_Dirac </P> 
602 
603 <P> Desde el punto de vista de la dinamica del cuerpo rigido, para nosotros el impulso es un cambio instantaneo en el vector
604  de velocidad, (es decir directamente cambiamos la velocidad sin tener que pasar por la aceleracion)
605  En la vida real los cambios de velocidad se dan apartir de una aceleracion, producto de aplicar una fuerza, segun
606  la segunda ley de movimiento de Newton</P>
607  
608 <P>  F = ma </P> 
609 
610 <P>  Sin embargo, el tiempo de aplicacion de la fuerza es menor que el elapsed_time de la simulacion, con lo cual 
611  numericamente, es preferible modificar directamente la velocidad.
612  El impulso produce un cambio en la velocidad (traslacion) del cuerpo rigido, que es inversamente proporcional a la masa. </P> 
613  
614 <P> -------------------------------------------------------------</P> 
615 <P> ecuacion (1)                 F = ma</P>
616 <P> -------------------------------------------------------------</P> 
617 
618 <P>  Si el impulso se produce en alguna linea de accion que pasa por el centro de masas (CM), solo se produce una movimiento 
619  de traslacion. Pero en cualquier otro eje de acciÃ³n, la fuerza tiene un cierto "momento" 
620  El momento de la fuerza con respecto al eje, es el responsable de un cambio en el movimiento de rotacion. </P> 
621  
622 <P>  http:*es.wikipedia.org/wiki/Momento_de_fuerza</P> 
623 
624 <P> En  terminos simples el momento de la fuerza es tanto mayor cuanto mas alejado ese del eje que pasa sobre el CM
625  Por ejemplo, si queremos abrir una puerta aplicamos una fuerza sobre la manija. Si quisieramos abrir la puerta aplicando
626  esa misma fuerza sobre el centro de la fuerza, nos costaria mucho mas.</P>
627  
628 <P> Por ultimo si quisieramos abrir la puerta aplicando la fuerza sobre la bisagra, no podriamos conseguir abrir la puerta
629  La fuerza aplicada sobre el eje que pasa por el CM no tiene momento (el momento es nulo). 
630 Si el momento de la fuerza es cero, la fuerza no produce rotacion.</P>

632 <P> Para computar el momento de la fuerza, tambien llamado torque, se usa el producto vectorial del vector posicion del
633  punto donde se aplica la fuerza,por el vector fuerza </P>
634 
635  <P>  T = r x F        T = torque, r = vector posicion, F = fuerza</P> 
636  
637 
638  <P> En nuestro caso el punto de aplicacion es contactPoint y el vector fuerza es impuse </P>
639 <P>  Torque = cross_vec(contactPoint, impulse)</P> 
640 
641 <P>  En 2 dimensiones, el producto cruz no es tan "intuitivo" como en 3d dimensiones. Es decir, no hay una forma 
642  standard de computar un cross product en 2d. 
643  En nuestro caso, lo que se precisa es la magnitud de lo que seria el cross product 3d si la coordenada Z fuerse cero
644  Esa magnitud es la que va a producir un cambio en al velocidad de rotacion, o velocidad angular. </P> 
645 
646 <P>  Ahora bien, no todos los torques producen la misma rotacion en los cuerpos rigidos, dependen de la resistencia que tenga
647  el cuerpo rigido de rotar sobre el eje particular. 
648  Ese concepto se llama momento de inercia, y es funciona de la misma forma que la masa para la aceleracion lineal.</P> 
649 
650 <P>  Cuando queremos mover un objeto con mas masa tenemos que aplicar mas fuerza. La masa se puede ver como la resistencia
651  que impone el cuerpo al movimiento lineal. Y como siempre esta descripta por la ecuacion de newton F = ma </P> 
652 
653 <P>  De la misma forma, cuando hablamos de movimientos de rotacion, el analogo a la fuerza es el "momento de la fuerza", o torque
654  y el analogo de la masa es el momento de inercia. El momento de inercia = I, depende no solo de la masa, (cuerpos mas masivos
655  son mas dificiles de hacer girar que cuerpos mas livianos), si no tambien ademas de la distribuicion de la masa
656  con respecto al eje sobre el que se quiere hacer girar. </P> 
657  
658 <P>  http://es.wikipedia.org/wiki/Momento_de_inercia </P> 
659 
660 <P> La ecuacion que describe estos 2 conceptos es </P>
661 <P>  -------------------------------------------------------------</P> 
662 <P> ecuacion (2)</P>
663 <P> T = I.dw/dt          T = torque, I = momento de inercia, dw/dt = aceleracion angular</P>
664 <P> -------------------------------------------------------------</P>
665 <P> http://es.wikipedia.org/wiki/Aceleraci%C3%B3n_angular</P>

667  * @method ApplyImpulse
668  * @param {Vector2} impulse impulso
669  * @param {Vector2} contactPoint punto de contacto
670 */
671 Body.prototype.ApplyImpulse = function (impulse, contactPoint)
672 {
673     if (this.inverseMass == 0) {
674         if (this.mass == 0)
675             return;     // es un cuerpo estatico
676             /*
677         else {
678             // si inverseMass = 0 pero mass >0 es porque era un cuerpo dinamico, que momentaneamente estaba 
679             // en equilibrio ( se comportaba como un cuerpo estatico)
680             // eso era valido mientras la unica fuerza que actuaba sobre el cuerpo es la gravedad
681             // Si le aplicamos un impulso, es porque hay otra fuerza, se rompio el equilibrio, y se vuelve a comportar como dinamico
682             this.inverseMass = 1 / this.mass;
683         }*/
684     }
685 
686 
687     var mJ = impulse.length();
688 
689     if (mJ > this.maxImpulse)
690         this.maxImpulse = mJ;
691 
692     if (this.inverseMass != 0 && this.maxForce != 0 && mJ > this.maxForce)
693     {
694         // el cuerpo revento, lo saco de la simulacion
695         this.deleted = true;
696         return;
697     }
698 
699 
700     // ecuacion (1)
701     // F = ma
702     this.velocity.add(vec2_multiply(impulse, this.inverseMass));
703 
704     // ecuacion (2)
705     // T = I.dw/dt          T = torque, I = momento de inercia, dw/dt = aceleracion angular
706     this.angularVelocity += this.inverseInertia * cross_vec(contactPoint, impulse);
707 
708 }

778 /**
779 <P> Ecuaciones de newton que tenemos que integrar para computar la posicion del cuerpo rigido</P>
780 <P>  Acceleration</P>
781 <P>     F = mA</P>
782 <P>  => A = F * 1/m</P>
783 <P>  Explicit Euler</P>
784 <P>  x += v * dt</P>
785 <P>  v += (1/m * F) * dt</P>
786 <P>  Symplectic Euler</P>
787 <P>  v += (1/m * F) * dt</P>
788 <P>  x += v * dt</P>
789 <P>  Para integrar conviene hacerlo en 2 etapas, avanzando el tiempo por la mitad. La explicacion se puede encontrar aca</P>
790 <P>  http://www.niksula.hut.fi/~hkankaan/Homepages/gravity.html</P>
791 <P>  se supone que dt = elapsed_time / 2</P>
792  * @method IntegrateForces
793  * @param {float} dt elapsed time / 2
794 */
795 Body.prototype.IntegrateForces = function( dt)
796 {
797     if(this.inverseMass == 0)
798         return;
799 
800     //	velocity += (force * inverseMass + gravity*gravity_factor) * dt;
801 
802     // aceleracion que producen las fuerzas externas
803     var acel = vec2_multiply(this.force, this.inverseMass);
804     // aceleracion de la gravedad
805     var acel_grav = vec2_multiply(gravity, this.gravity_factor);
806     // aceleracion neta
807     var acel_neta = vec2_add(acel, acel_grav);
808     // integro la aceleracion lineal
809     this.velocity.add(vec2_multiply(acel_neta, dt));
810     // integro la aceleracion angular
811     this.angularVelocity += this.torque * this.inverseInertia * dt;
812 
813 }
